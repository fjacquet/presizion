        scenario.diskPerServerGb,

    }

    // CALC-02: RAM-limited count
    const ramUtilPct = cluster.ramUtilizationPercent ?? 100;

    4. **CALC-01-GHZ modification** -- In the `sizingMode === 'ghz'` branch, deduct vSAN CPU overhead from the target server GHz capacity:
    ```typescript
    } else if (sizingMode === 'ghz') {
      const effectiveTargetGhzPerCore = scenario.vsanFttPolicy

      // ... existing serverCountByGhz call with effectiveTargetGhzPerCore
    ```
    WAIT -- the existing serverCountByGhz takes `targetCpuFrequencyGhz` (per-core) and `coresPerServer` separately. vSAN CPU overhead is a percentage of total node GHz. The cleanest approach: compute node-level GHz, apply overhead, then pass the effective per-core frequency back.
    ```typescript
    const rawTargetFreqGhz = scenario.targetCpuFrequencyGhz ?? 1;
    const effectiveTargetFreqGhz = scenario.vsanFttPolicy && scenario.vsanCpuOverheadPercent !== undefined
      ? rawTargetFreqGhz * (1 - (scenario.vsanCpuOverheadPercent) / 100)
      : scenario.vsanFttPolicy
        ? rawTargetFreqGhz * (1 - VSAN_DEFAULT_CPU_OVERHEAD_PCT / 100)  // default 10% when vSAN active
        : rawTargetFreqGhz;  // no vSAN, no deduction
    ```
    Actually, simplify: when vsanFttPolicy is set, always apply overhead (default 10%). When absent, no deduction.
    ```typescript
    const rawTargetFreqGhz = scenario.targetCpuFrequencyGhz ?? 1;
    const effectiveTargetFreqGhz = scenario.vsanFttPolicy
      ? computeVsanEffectiveGhzPerNode(rawTargetFreqGhz, scenario.vsanCpuOverheadPercent)
      : rawTargetFreqGhz;
    cpuLimitedCount = serverCountByGhz(
      cluster.totalPcores,
      cluster.cpuFrequencyGhz ?? 1,
      cpuUtilPct,
      headroomFactor,
      effectiveTargetFreqGhz,
      coresPerServer,
      targetCpuUtilPct,
    );
    ```
    Note: `computeVsanEffectiveGhzPerNode` already uses `VSAN_DEFAULT_CPU_OVERHEAD_PCT` as the default for the second parameter. Per Research open question #3, vSAN CPU overhead only applies in GHz mode. In vcpu/specint/aggressive modes, no vSAN CPU deduction.

For now, keep the existing simple ratio as-is (raw demand / raw capacity) since the breakdown tables (Phase 19) will handle the detailed accounting.
