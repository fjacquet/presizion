       ```
       const isMultiScope = !isJson

         && result.detectedScopes.length > 1

       const [selectedScopes, setSelectedScopes] = useState<string[]>(
         () => result.detectedScopes ?? []
       )

       // Re-initialize when result changes (new import)
       useEffect(() => {
         setSelectedScopes(result.detectedScopes ?? [])
       }, [result])

       // Live-preview cluster: re-aggregate if multi-scope and rawByScope available, otherwise use result directly
       const previewCluster = isMultiScope && result.rawByScope != null
         ? aggregateScopes(result.rawByScope, selectedScopes)
         : result
       ```

    4. Add scope selector section in JSX — insert ABOVE the existing preview data rows `<div className="space-y-1 text-sm">`:
       ```jsx
       {isMultiScope && result.scopeLabels != null && (
         <div className="space-y-2">
           <p className="text-sm font-medium">Filter by cluster</p>
           <div className="space-y-1">
             {result.detectedScopes!.map((key) => (
               <div key={key} className="flex items-center gap-2">
                 <Checkbox
                   id={`scope-${key}`}
                   checked={selectedScopes.includes(key)}
                   onCheckedChange={(checked) => {
                     setSelectedScopes((prev) =>
                       checked
                         ? [...prev, key]
                         : prev.filter((k) => k !== key)
                     )
                   }}
                 />
                 <label htmlFor={`scope-${key}`} className="text-sm cursor-pointer">
                   {result.scopeLabels![key] ?? key}
                 </label>
               </div>
             ))}
           </div>
         </div>
       )}
       ```
