```typescript
/**
 * Capacity breakdown for a single resource (CPU or Memory).
 *
 * Invariant (CAP-06): required + spare + excess === total
 * where spare = reservedMaxUtil + haReserve
 */
export interface ResourceBreakdown {
  readonly vmsRequired: number;       // demand from VMs (after growth, before overhead)
  readonly vsanConsumption: number;   // vSAN overhead contribution (0 when no vSAN)
  readonly required: number;          // vmsRequired + vsanConsumption
  readonly reservedMaxUtil: number;   // required / maxUtilPct * (1 - maxUtilPct) (CAP-05)
  readonly haReserve: number;         // one host worth of capacity (CAP-04)
  readonly spare: number;             // reservedMaxUtil + haReserve
  readonly excess: number;            // total - required - spare (can be negative)
  readonly total: number;             // finalCount * capacityPerNode
}

/**
 * Storage capacity breakdown with additional vSAN decomposition rows.
 * Extends ResourceBreakdown; all values in GiB internally (VSAN-10).
 *
 * Invariant (CAP-06): required + spare + excess === total
 * where required = rawRequired, spare = slackSpace + haReserve
 */
export interface StorageBreakdown extends ResourceBreakdown {
  readonly usableRequired: number;    // VM usable storage demand in GiB
  readonly swapOverhead: number;      // VM swap space in GiB (0 when disabled)
  readonly metadataOverhead: number;  // vSAN metadata in GiB
  readonly fttOverhead: number;       // extra copies due to FTT multiplier
  readonly rawRequired: number;       // total raw storage after pipeline
  readonly slackSpace: number;        // vSAN slack reserve in GiB
}

/**
 * Complete capacity breakdown for one scenario.
 * Produced by computeVsanBreakdown().
 */
export interface VsanCapacityBreakdown {
  readonly scenarioId: string;
  readonly cpu: ResourceBreakdown;
  readonly memory: ResourceBreakdown;
  readonly storage: StorageBreakdown;
  readonly minNodesByConstraint: Record<string, number>;
}
```

Create `src/lib/sizing/vsanBreakdown.ts` with a single exported function:

```typescript
export function computeVsanBreakdown(
  cluster: OldCluster,
  scenario: Scenario,
  result: ScenarioResult,
): VsanCapacityBreakdown
```

The function accepts a pre-computed ScenarioResult (NOT recomputed internally -- Pitfall 6 from research). It derives all breakdown values using the same demand figures that computeScenarioResult used:

CPU breakdown:
- Compute demand from cluster/scenario: For GHz reporting, convert vCPU demand to GHz using `totalVcpus * (scenario.targetCpuFrequencyGhz ?? 1)`. For growth-aware demand, use the same growth factor formula `(1 + (scenario.cpuGrowthPercent ?? 0) / 100)`.
  NOTE: This function reads growth fields from scenario to reconstruct the grown demand. The ScenarioResult already reflects growth but we need the breakdown of the demand components.
- vmsRequired = grown CPU demand in GHz
- vsanConsumption = totalConfiguredGhz * (vsanCpuOverheadPct / 100) when vSAN active, else 0
- required = vmsRequired + vsanConsumption
- totalConfiguredGhz = finalCount * coresPerServer * freqGhz
- HA reserve = one node GHz = coresPerServer * freqGhz
- reservedMaxUtil per CAP-05 formula
- spare = reservedMaxUtil + haReserve
- excess = total - required - spare

Memory breakdown:
- vmsRequired = effectiveVmCount * ramPerVmGb * memGrowthFactor * (ramUtilPct / 100)
- vsanConsumption = finalCount * vsanMemoryPerHostGb when vSAN active, else 0
- total = finalCount * ramPerServerGb

Storage breakdown:
- Non-vSAN: simple usableRequired = effectiveVmCount * diskPerVmGb * storageGrowthFactor, rawRequired = usableRequired
- vSAN: reuse computeVsanStorageRaw for rawRequired, then decompose to get fttOverhead, metadata, slack components
- haReserve = total / finalCount (1/N of cluster)
- For storage: spare = slackSpace + haReserve (slack is part of spare since it is already included in rawRequired from the pipeline, but we separate it for display)

IMPORTANT nuance for storage invariant: The rawRequired from computeVsanStorageRaw INCLUDES slack. For the invariant to hold:
- required = rawRequired (the total raw demand including FTT + metadata + slack)
  WAIT -- this means slack would be double-counted in spare. Need to think this through.
  Correct approach: required = raw BEFORE slack (= afterFtt + metadata). Then slack is part of spare.
  So: required = rawBeforeSlack, spare = slackSpace + haReserve, excess = total - required - spare.
  The invariant: rawBeforeSlack + slackSpace + haReserve + excess = total.

Actually, let me align with CAP-03 requirements: "Total Raw Required" includes FTT + metadata but the table separates slack. So:
- required = rawBeforeSlack (FTT + metadata, no slack)
- spare = slackSpace + haReserve
- excess = total - required - spare
- rawRequired = the full pipeline output (includes slack) for informational display

This way the invariant required + spare + excess = total holds cleanly.

minNodesByConstraint:
- cpu: result.cpuLimitedCount
- memory: result.ramLimitedCount
- storage: result.diskLimitedCount
- ftha: result.haReserveCount (N+1/N+2 added servers)
- vms: ceil(effectiveVmCount / result.vmsPerServer) if vmsPerServer > 0, else 0
